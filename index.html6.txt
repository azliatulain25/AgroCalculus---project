If we apply potassium (K) fertilizer too early, it is lost through heavy tropical leaching; if applied too late, the cell elongation phase has already concluded, resulting in small, sour fruits susceptible to sunburn and internal "rot/core browning" diseases.

        <!-- Tab 2: Math -->
        <div id="tab-math" class="tab-content hidden space-y-4">
          <h4 class="font-bold text-slate-900 text-lg">Mathematical Derivation of Fruit Growth Dynamics</h4>
          <p class="text-slate-600 text-sm leading-relaxed">
            The growth profile of many organic fruits resembles an S-shaped curve modeled by the classic <strong>Verhulst Logistic Equation</strong>:
          </p>
          <div class="bg-slate-50 p-4 rounded-xl border border-slate-100 overflow-x-auto text-sm sm:text-base font-mono">
            $$W(t) = \frac{L}{1 + e^{-k(t - t_0)}}$$
          </div>
          <p class="text-slate-600 text-sm leading-relaxed">
            Where \(L\) is the ultimate carrying capacity (mature weight), \(k\) is the growth constant governing the slope steepness, and \(t_0\) is the midpoint inflection time.
          </p>
          <p class="text-slate-600 text-sm leading-relaxed font-semibold">
            Using the Chain Rule and Quotient Rule, we derive the rate of change \(W'(t)\):
          </p>
          <div class="bg-slate-50 p-4 rounded-xl border border-slate-100 overflow-x-auto text-sm sm:text-base font-mono space-y-2 text-slate-700">
            <p>1. Rewrite the function with negative exponents: \(W(t) = L \cdot \left(1 + e^{-k(t-t_0)}\right)^{-1}\)</p>
            <p>2. Differentiate with respect to time \(t\):</p>
            <p class="pl-4">\(W'(t) = -L \cdot \left(1 + e^{-k(t-t_0)}\right)^{-2} \cdot \frac{d}{dt}\left[1 + e^{-k(t-t_0)}\right]\)</p>
            <p class="pl-4">\(W'(t) = -L \cdot \left(1 + e^{-k(t-t_0)}\right)^{-2} \cdot \left(-k \cdot e^{-k(t-t_0)}\right)\)</p>
            <p class="pl-4">\(W'(t) = \frac{L \cdot k \cdot e^{-k(t-t_0)}}{\left(1 + e^{-k(t-t_0)}\right)^2}\)</p>
            <p>3. This analytical derivative represents the exact slope of the tangent line drawn on our graph.</p>
          </div>
          <p class="text-slate-600 text-sm leading-relaxed">
            By analyzing where the second derivative \(W''(t) = 0\), we discover the point of maximum velocity—the <strong>Inflection Point</strong>—which falls exactly at \(t = t_0\) (Week 15 in this simulator).
          </p>
        </div>

    // --- Model Constants ---
    const L = 2100;    // Carried carrying capacity of mature pineapple (grams)
    const k = 0.45;    // Growth speed constant (week^-1)
    const t0 = 15;     // Midpoint week where growth peaks (weeks post forcing)
    const MIN_WEEK = 8;
    const MAX_WEEK = 22;

    // --- State Variables ---
    let currentWeek = 15.0; // Starts at midpoint week
    let growthChartInstance = null;

    // --- Core Mathematics functions ---
    
    // Calculates cumulative weight W(t)
    function W(t) {
      return L / (1 + Math.exp(-k * (t - t0)));
    }

    // Analytical derivative W'(t) representing instantaneous growth velocity
    function W_prime(t) {
      const wt = W(t);
      // W'(t) = k * W(t) * (1 - W(t)/L)
      return k * wt * (1 - (wt / L));
    }

    // Calculates Y-coordinate on the tangent line for a given x, based on a pivot week ts
    function getTangentY(x, ts) {
      const w_s = W(ts);
      const wp_s = W_prime(ts);
      return wp_s * (x - ts) + w_s;
    }

    // --- Generate Dataset Points for Charting ---
    function generateDatasetPoints() {
      const points = [];
      for (let t = MIN_WEEK; t <= MAX_WEEK; t += 0.25) {
        points.push({ x: t, y: W(t) });
      }
      return points;
    }

    // Generate dynamic tangent line points based on the selected week
    function generateTangentPoints(ts) {
      const points = [];
      
      // We draw the tangent line extending +/- 2.5 weeks on either side of selected week to keep it relative
      let t_start = Math.max(MIN_WEEK, ts - 2.5);
      let t_end = Math.min(MAX_WEEK, ts + 2.5);

      let y_start = getTangentY(t_start, ts);
      let y_end = getTangentY(t_end, ts);

      // Guard against drawing the tangent line below zero weight (unbiological)
      if (y_start < 0) {
        // Solve for t when y = 0: wp * (t - ts) + ws = 0 => t = ts - ws / wp
        const zero_intersect = ts - (W(ts) / W_prime(ts));
        t_start = Math.max(MIN_WEEK, zero_intersect);
        y_start = 0;
      }

      points.push({ x: t_start, y: y_start });
      points.push({ x: t_end, y: y_end });

      return points;
    }

    // --- Core UI Update Loop ---
    function updateSimulation(val) {
      currentWeek = parseFloat(val);
      
      // Calculations
      const weightVal = W(currentWeek);
      const derivVal = W_prime(currentWeek);
      const dailyDeriv = derivVal / 7;
      const pctOfMax = (weightVal / L) * 100;

      // Update Slider Visual text
      document.getElementById('slider-val').textContent = `${currentWeek.toFixed(1)} weeks`;

      // Update Calculus Output Cards
      document.getElementById('out-week').textContent = currentWeek.toFixed(1);
      document.getElementById('out-weight').textContent = weightVal.toFixed(1);
      document.getElementById('out-pct').textContent = `${pctOfMax.toFixed(1)}% of maximum potential weight achieved`;
      document.getElementById('out-deriv').textContent = derivVal.toFixed(2);
      document.getElementById('out-daily-deriv').textContent = `(~${dailyDeriv.toFixed(1)} grams/day mass accumulation)`;

      // Update Chart.js dataset properties
      if (growthChartInstance) {
        // Update Tangent dataset
        growthChartInstance.data.datasets[1].data = generateTangentPoints(currentWeek);
        // Update Pivot point dataset
        growthChartInstance.data.datasets[2].data = [{ x: currentWeek, y: weightVal }];
        // Redraw without animation for instantaneous dragging feedback
        growthChartInstance.update('none');
      }

      // Update Dynamic Layman Explanations
      const phaseInfo = getPhaseData(currentWeek);
      const expCard = document.getElementById('explanation-card');
      const expIcon = document.getElementById('explanation-icon');
      const statusTag = document.getElementById('status-tag');

      // Change card classes based on phase colors
      expCard.className = `rounded-2xl p-6 border transition-all duration-300 shadow-sm ${phaseInfo.color}`;
      expIcon.className = `p-3 rounded-xl shadow-inner mt-1 ${phaseInfo.accentClass}`;
      expIcon.innerHTML = phaseInfo.svg;
      
      document.getElementById('phase-title').textContent = phaseInfo.title;
      document.getElementById('phase-explanation').textContent = phaseInfo.explanation;
      document.getElementById('phase-action').textContent = phaseInfo.action;

      statusTag.textContent = phaseInfo.title.split(' (')[0];
      statusTag.className = `text-xs font-bold px-2.5 py-1 rounded-full self-start sm:self-auto shadow-sm ${phaseInfo.accentClass}`;

      // --- MathJax / KaTeX Equation Renders ---
      const mathGrowthHTML = `W(${currentWeek.toFixed(1)}) = \\frac{2100}{1 + e^{-0.45(${currentWeek.toFixed(1)} - 15)}} = ${weightVal.toFixed(1)}\\text{ g}`;
      
      const mathDerivHTML = `
        \\begin{aligned}
        W'(${currentWeek.toFixed(1)}) &= 0.45 \\cdot W(${currentWeek.toFixed(1)}) \\cdot \\left(1 - \\frac{W(${currentWeek.toFixed(1)})}{2100}\\right) \\\\
        &= 0.45 \\cdot ${weightVal.toFixed(0)} \\cdot \\left(1 - \\frac{${weightVal.toFixed(0)}}{2100}\\right) \\\\
        &= ${derivVal.toFixed(2)}\\text{ g/week}
        \\end{aligned}
      `;

      // Render math blocks
      katex.render(mathGrowthHTML, document.getElementById('growth-math-display'), { displayMode: true, throwOnError: false });
      katex.render(mathDerivHTML, document.getElementById('deriv-math-display'), { displayMode: true, throwOnError: false });
    }

      // Highlight active button
